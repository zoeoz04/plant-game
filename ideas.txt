<!DOCTYPE html>
<html lang="en">
<head>
<meta charset="UTF-8">
<meta name="viewport" content="width=device-width, initial-scale=1.0">
<title>Plant Bug Clicker</title>
<style>
    /* CSS styles for the plant and bugs */
    #plant {
        width: 100px;
        height: 100px;
        background-color: green;
        position: relative;
    }
    .bug {
        width: 20px;
        height: 20px;
        background-color: red;
        border-radius: 50%;
        position: absolute;
    }
</style>
</head>
<body>
<div id="plant"></div>
<script>
    // JavaScript code to handle bug generation and removal
    const plant = document.getElementById('plant');

    function generateBug() {
        const bug = document.createElement('div');
        bug.className = 'bug';
        bug.style.left = Math.random() * (plant.clientWidth - 20) + 'px';
        bug.style.top = Math.random() * (plant.clientHeight - 20) + 'px';
        plant.appendChild(bug);

        // Remove bug after a set time
        setTimeout(() => {
            plant.removeChild(bug);
        }, 2000); // Adjust this time as needed
    }

    // Generate bugs every 5 seconds
    setInterval(generateBug, 5000);

    // Remove bug when clicked
    plant.addEventListener('click', (event) => {
        if (event.target.classList.contains('bug')) {
            plant.removeChild(event.target);
        }
    });
</script>
</body>
</html>
